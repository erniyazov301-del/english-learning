<script>
    let state = JSON.parse(localStorage.getItem('eng_app')) || { setup: false, level: 'A1', index: 0, font: '20px', known: [] };

    const content = {
        A1: { s: ["I drink coffee", "He is a doctor", "Where is the station?"], w: ["Apple", "House", "Friend"] },
        B2: { s: ["I have been waiting for hours", "Despite the rain, we went out"], w: ["Incentive", "Paradigm", "Leverage"] },
        C2: { s: ["The implications are far-reaching", "He articulated his views eloquently"], w: ["Ethereal", "Quintessential", "Sagacity"] }
    };

    const grammar = [
        { t: "Present Simple", d: "Habits and facts.", e: "She works every day." },
        { t: "Passive Voice", d: "Focus on the object.", e: "The letter was sent." }
    ];

    function init() {
        if (!state.setup) {
            document.getElementById('setup-screen').classList.remove('hidden');
        } else {
            document.body.style.fontSize = state.font;
            render();
            renderGrammar();
        }
    }

    function saveSetup() {
        state.level = document.getElementById('lvl-select').value;
        state.font = document.getElementById('font-select').value;
        state.setup = true;
        save();
        location.reload();
    }

    function tab(name) {
        ['tests', 'cards', 'grammar'].forEach(t => {
            document.getElementById('view-' + t).classList.add('hidden');
            document.getElementById('nav-' + t).classList.remove('active');
        });
        document.getElementById('view-' + name).classList.remove('hidden');
        document.getElementById('nav-' + name).classList.add('active');
    }

    function render() {
        const pool = content[state.level].s;
        const cur = pool[state.index % pool.length];
        document.getElementById('current-sentence').innerText = cur;
        document.getElementById('p-text').innerText = `Sentence ${state.index + 1} / 500`;
        document.getElementById('p-bar').style.width = `${(state.index / 500) * 100}%`;
        
        const words = content[state.level].w;
        document.getElementById('card-text').innerText = words[0];
    }

    function check() {
        const input = document.getElementById('answer-input').value.toLowerCase().trim();
        const feedback = document.getElementById('feedback');
        
        // Demo logic: in a real app, you'd have a mapping. 
        // Here we just accept "test" as correct for the demo.
        if(input.length > 0) {
            feedback.innerText = "✓ Correct";
            feedback.style.color = "green";
            setTimeout(() => {
                state.index++;
                save();
                document.getElementById('answer-input').value = "";
                feedback.innerText = "";
                render();
            }, 1000);
        }
    }

    function nextCard(known) {
        const words = content[state.level].w;
        const word = words.shift();
        if(!known) words.push(word);
        document.getElementById('card-text').innerText = words[0] || "Finished!";
    }

    function renderGrammar() {
        const html = grammar.map(g => `
            <div class="lesson">
                <h3>${g.t}</h3>
                <p>${g.d}</p>
                <div class="example">${g.e}</div>
            </div>
        `).join('');
        document.getElementById('grammar-list').innerHTML = html;
    }

    function handleKey(e) { if(e.key === 'Enter') check(); }
    function save() { localStorage.setItem('eng_app', JSON.stringify(state)); }

    init();
</script>
